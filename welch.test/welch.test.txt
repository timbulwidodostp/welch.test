# Welch's Heteroscedastic F Test and Welch's Heteroscedastic F Test with Trimmed Means and Winsorized Variances Use welch.test (onewaytests) With (In) R Software
install.packages("onewaytests")
library("onewaytests")
# Estimation Welch's Heteroscedastic F Test and Welch's Heteroscedastic F Test with Trimmed Means and Winsorized Variances Use welch.test (onewaytests) With (In) R Software
welch.test = read.csv("https://raw.githubusercontent.com/timbulwidodostp/welch.test/main/welch.test/welch.test.csv", sep = ";")
welch.test <- welch.test(welch.test ~ welch.test_, data = welch.test)
summary(welch.test)
paircomp(welch.test)
